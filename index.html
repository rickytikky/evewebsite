    <script>

        const uploadArea = document.getElementById('uploadArea');

        const fileInput = document.getElementById('fileInput');

        const selectedFile = document.getElementById('selectedFile');

        const fileName = document.getElementById('fileName');

        const fileSize = document.getElementById('fileSize');

        const uploadSection = document.getElementById('uploadSection');

        const analysisSection = document.getElementById('analysisSection');

        const overviewSection = document.getElementById('overviewSection');

        const statusText = document.getElementById('statusText');

        const statusSubtext = document.getElementById('statusSubtext');


        // Drag and drop handlers

        uploadArea.addEventListener('dragover', (e) => {

            e.preventDefault();

            uploadArea.classList.add('dragover');

        });


        uploadArea.addEventListener('dragleave', () => {

            uploadArea.classList.remove('dragover');

        });


        uploadArea.addEventListener('drop', (e) => {

            e.preventDefault();

            uploadArea.classList.remove('dragover');

            const files = e.dataTransfer.files;

            if (files.length > 0) {

                handleFile(files[0]);

            }

        });


        uploadArea.addEventListener('click', () => {

            fileInput.click();

        });


        fileInput.addEventListener('change', (e) => {

            if (e.target.files.length > 0) {

                handleFile(e.target.files[0]);

            }

        });


        function handleFile(file) {

            fileName.textContent = file.name;

            fileSize.textContent = formatFileSize(file.size);

            selectedFile.classList.add('show');


            // Start analysis after a brief delay

            setTimeout(() => {

                startAnalysis();

            }, 1000);

        }


        function formatFileSize(bytes) {

            if (bytes === 0) return '0 Bytes';

            const k = 1024;

            const sizes = ['Bytes', 'KB', 'MB', 'GB'];

            const i = Math.floor(Math.log(bytes) / Math.log(k));

            return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i];

        }


        function startAnalysis() {

            uploadSection.classList.add('hidden');

            analysisSection.classList.add('show');


            // Simulate analysis phases

            const phases = [

                { text: 'Analyzing file structure...', subtext: 'Reading and parsing document contents' },

                { text: 'Extracting data points...', subtext: 'Identifying relevant information' },

                { text: 'Categorizing information...', subtext: 'Sorting into life markers' },

                { text: 'Processing categories...', subtext: 'Finance, Health, Social, Other' },

                { text: 'Generating insights...', subtext: 'Creating cross-analysis and overview' }

            ];


            let phaseIndex = 0;

            const phaseInterval = setInterval(() => {

                if (phaseIndex < phases.length) {

                    statusText.textContent = phases[phaseIndex].text;

                    statusSubtext.textContent = phases[phaseIndex].subtext;

                    phaseIndex++;

                } else {

                    clearInterval(phaseInterval);

                    setTimeout(() => {

                        showCategories();

                    }, 1000);

                }

            }, 1500);

        }

                social: {

                    percentage: 68

                },

        function showOverview(categories) {

            analysisSection.classList.remove('show');

            overviewSection.classList.add('show');


            // Calculate statistics

            const totalItems = Object.values(categories).reduce((sum, cat) => sum + cat.items.length, 0);

            const avgPercentage = Math.round(

                Object.values(categories).reduce((sum, cat) => sum + cat.percentage, 0) / 

                Object.keys(categories).length

            );


            // Animate stats

            animateValue('totalItems', 0, totalItems, 1000);

            animateValue('categoriesFound', 0, Object.keys(categories).length, 1000);

            animateValue('completionScore', 0, avgPercentage, 1000, '%');


            // Update insight texts with dynamic content

            const diagnosticText = document.getElementById('diagnosticText');

            diagnosticText.textContent = `Your file has been thoroughly analyzed across ${Object.keys(categories).length} life marker categories. 

                We've identified ${totalItems} data points that provide insights into your financial patterns (${categories.finance.percentage}% complete), 

                health indicators (${categories.health.percentage}% complete), and social connections (${categories.social.percentage}% complete).`;


            const crossAnalysisText = document.getElementById('crossAnalysisText');

            crossAnalysisText.textContent = `Cross-referencing reveals interesting patterns: your financial activity shows ${categories.finance.items.length} key areas, 

                while health data indicates ${categories.health.items.length} wellness markers. Social connections are tracking ${categories.social.items.length} 

                relationship insights. These correlations help provide a holistic view of your life balance.`;


            const recommendationsText = document.getElementById('recommendationsText');

            recommendationsText.textContent = `Based on the analysis, consider optimizing your ${categories.finance.percentage > 75 ? 'highly active' : 'developing'} financial profile, 

                ${categories.health.percentage > 70 ? 'maintaining' : 'improving'} your health metrics, and 

                ${categories.social.percentage > 65 ? 'continuing to nurture' : 'expanding'} your social connections for better life balance.`;

        }


        function animateValue(id, start, end, duration, suffix = '') {

            const element = document.getElementById(id);

            const range = end - start;

            const increment = range / (duration / 16);

            let current = start;


            const timer = setInterval(() => {

                current += increment;

                if ((increment > 0 && current >= end) || (increment < 0 && current <= end)) {

                    element.textContent = end + suffix;

                    clearInterval(timer);

                } else {

                    element.textContent = Math.round(current) + suffix;

                }

            }, 16);

        }

        }
